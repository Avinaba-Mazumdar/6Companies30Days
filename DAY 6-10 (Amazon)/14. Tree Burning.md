class Solution {
    public:int minTime(Node* root, int target) {
        if(!root) return -1;
        unordered_map<Node*, Node*> map;
        Node* trg = mapParents(map, root, target);
        
        return maxDist(map, trg);
    }
    
    private:
    Node* mapParents(unordered_map<Node*, Node*> &map, Node* root, int target){
        queue<Node*> Q; Q.push(root);
        Node *ans;
        
        while(!Q.empty()){
            int n = Q.size();
            
            while(n--){
                Node *node = Q.front(); Q.pop();
                
                if(node->left){
                    Q.push(node->left);
                    map[node->left] = node;
                }
                if(node->right){
                    Q.push(node->right);
                    map[node->right] = node;
                }
                if(node->data == target) ans = node;
            }
        }
        
        return ans;
    }
    
    int maxDist(unordered_map<Node*, Node*> map, Node* trg){
        queue<Node*> Q; Q.push(trg);
        unordered_map<Node*, int> vis;
        vis[trg] = 1;
        int ans = 0;
        
        while(!Q.empty()){
            int n = Q.size();
            while(n--){
                Node* node = Q.front(); Q.pop();
                
                if(node->left and !vis[node->left]){
                    vis[node->left] = 1;
                    Q.push(node->left);
                }
                if(node->right and !vis[node->right]){
                    vis[node->right] = 1;
                    Q.push(node->right);
                }
                if(map[node] and !vis[map[node]]){
                    vis[map[node]] = 1;
                    Q.push(map[node]);
                }
            }
            if(Q.size()) ans++;
        }
        
        return ans;
    }
};
